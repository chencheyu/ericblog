def gcd(p,q):
  if (q == 0):
    return ( p, 1, 0 )
  d,b,a = gcd(q, p % q);
  b=b-(p//q)*a
  return (d, a, b)
  

def inverse(k,n):
    d,a,b= gcd(k, n);
    if (d > 1):
      print("Inverse does not exist."); return 0;
    if (a > 0): 
      return a;
    return n + a;

def keygen(p,q): 
  n=p*q
  oln=(q-1)*(p-1)
  e=prim[random.randint(0,len(prim))]
  d=inverse(e,oln)
  return (e,n),(d,n)

def decode(s):
  #english only
  ls=[]
  if len(s)%2==1:
    s+=' '
  for i in range(0,len(s),2):
    ls.append(ord(s[i])*0xff+ord(s[i+1]))
  return ls

def encode(ls):
  s=""
  for i in ls:
    s=s+chr(i//0xff)+chr(i%0xff)
  return s

def syc(m,e,n):
  c=[]
  for i in range(0,len(m)):
    c.append(pow(m[i],e,n))
    
  return c

def usyc(c,d,n):
  m=[]
  for i in range(0,len(c)):
    m.append(pow(c[i],d,n))
    
  return m
